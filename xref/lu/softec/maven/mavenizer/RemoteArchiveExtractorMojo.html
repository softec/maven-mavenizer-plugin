32  /**
33   * Download and extract the binaries of the project
34   *
35   * @goal download
36   * @phase generate-sources
37   */
38  public class RemoteArchiveExtractorMojo extends AbstractArchiveMavenizerMojo
39  {
40      /**
41       * The Plexus Archiver Manager.
42       *
43       * @component role="org.codehaus.plexus.archiver.manager.ArchiverManager"
44       */
45      private ArchiverManager archiveManager;
46  
47      /**
48       * @throws MojoExecutionException if the operation fails, even partially
49       */
50      public void execute() throws MojoExecutionException
51      {
52          // If no archive file is provided, expect the binaries to be already extracted and do nothing
53          if (getArchiveFile() == null) {
54              return;
55          }
56  
57          // Retrieve archive if URL is provided and remote file is newer than local one
58  
59          if (getURLLastModificationDate() > getArchiveFile().lastModified()) {
60              if (!isBuilt(getArchiveFile()) || !isBuilt(getBinariesBaseDir())) {
61                  if (!isBuilt(getBinariesBaseDir())) {
62                      getLog()
63                          .warn(
64                              "Newer archive file has not been downloaded because binaries are not stored in the build directory and should therefore be updated manually.");
65                  } else {
66                      getLog()
67                          .warn(
68                              "Newer archive file has not been downloaded because it is not stored in the build directory and should therefore be updated manually.");
69                  }
70              } else {
71                  getLog().info(
72                      "Retrieving project archive from " + getArchiveURL().toString() + " to " +
73                          getArchiveFile().getAbsolutePath());
74  
75                  try {
76                      FileUtils.copyURLToFile(getArchiveURL(), getArchiveFile());
77                  } catch (IOException e) {
78                      throw new MojoExecutionException(
79                          "Unable to retrieve the project archive from " + getArchiveURL().toString() + " to " +
80                              getArchiveFile().getAbsolutePath(), e);
81                  }
82  
83                  forceDeleteTargetDirectory(getBinariesBaseDir());
84              }
85          } else {
86              if (getArchiveURL() != null && isBuilt(getArchiveFile()) && isBuilt(getBinariesBaseDir())) {
87                  getLog().info("Remote file is not newer, skipping archive download.");
88              } else {
89                  if (isBuilt(getBinariesBaseDir())) {
90                      getLog().info("Archive should be provided locally, skipping download.");
91                  } else {
92                      getLog().info("Binaries should be provided locally, skipping download.");
93                  }
94              }
95          }
96  
97          // Extract archive file if archive is newer than extracted data
98  
99          if (getArchiveFile().lastModified() > getBinariesBaseDir().lastModified()) {
100             if (!isBuilt(getBinariesBaseDir())) {
101                 getLog()
102                     .warn(
103                         "Newer archive file has not been extracted because the binary base directory it is not stored in the build directory");
104             } else {
105                 getBinariesBaseDir().mkdirs();
106                 try {
107                     UnArchiver unarchiver = archiveManager.getUnArchiver(getArchiveFile());
108                     unarchiver.setSourceFile(getArchiveFile());
109                     unarchiver.setDestDirectory(getBinariesBaseDir());
110 
111                     // Extract libraries
112                     FileSelector selector = getLibsSelector();
113                     if (selector != null) {
114                         unarchiver.setFileSelectors(new FileSelector[]{selector});
115                         getLog().info("Expanding libraries from project archive " + getArchiveFile().getAbsolutePath());
116                     } else {
117                         getLog().info("Expanding all files from project archive " + getArchiveFile().getAbsolutePath());
118                     }
119                     unarchiver.extract();
120 
121                     // If libraries has been selectively extracted and other dependencies are selected, extract them
122                     if (selector != null) {
123                         selector = getDepsSelector();
124                         if (selector != null) {
125                             unarchiver.setFileSelectors(new FileSelector[]{selector});
126                             getLog()
127                                 .info("Expanding dependencies from project archive " +
128                                     getArchiveFile().getAbsolutePath());
129                             unarchiver.extract();
130                         }
131                     }
132                 } catch (NoSuchArchiverException e) {
133                     forceDeleteTargetDirectory(getBinariesBaseDir());
134                     throw new MojoExecutionException("No unarchiver is available to handle the retrieved archive", e);
135                 } catch (ArchiverException e) {
136                     forceDeleteTargetDirectory(getBinariesBaseDir());
137                     throw new MojoExecutionException("Unable to expand the retrieved archive", e);
138                 }
139             }
140         } else {
141             if (isBuilt(getBinariesBaseDir())) {
142                 getLog().info("Archive file is not newer, skipping archive expansion.");
143             } else {
144                 getLog().info("Binaries should be provided locally, skipping archive extraction.");
145             }
146         }
147     }

161      * Set the URL used to check and download the latest version of the archive file of the project to mavenize.
162      *

166     public void setArchiveURL(URL archiveURL)

168         this.archiveURL = archiveURL;
169     }

171     /**
172      * Return a FileSelector for selecting the libraries from the archive file. If no selection has been provided, null
173      * is returned.
174      *
175      * @return a FileSelector for selecting the libraries into the archive file.
176      */
177     public FileSelector getLibsSelector()
178     {
179         return getSelector(getLibsIncludes(), getLibsExcludes());
180     }
181 
182     /**
183      * Return a FileSelector for selecting the additionnal libraries used only for dependency resolution from the
184      * archive file.  If no selection has been provided, null is returned.
185      *
186      * @return a FileSelector for selecting the additionnal libraries used only for dependency resolution from the
187      *         archive file.
188      */
189     public FileSelector getDepsSelector()
190     {
191         return getSelector(getDepsIncludes(), getDepsExcludes());
192     }
193 
194     /**
195      * Internal helper function to build IncludeExcludeFileSelector
196      *
197      * @param includes includes patterns for the selector
198      * @param excludes excludes patterns for the selector
199      * @return a IncludeExcludeFileSelector based on the includes and excludes arguments. If both arguments are null,
200      *         null is returned.
201      */
202     private static FileSelector getSelector(String[] includes, String[] excludes)
203     {
204         if (includes == null && excludes == null) {
205             return null;
206         }
207 
208         IncludeExcludeFileSelector fs = new IncludeExcludeFileSelector();
209         fs.setIncludes(includes);
210         fs.setExcludes(excludes);
211         fs.setUseDefaultExcludes(true);
212         return fs;
213     }
214 
215     /**
216      * Return the timestamp of the last modification to the remote archive file. If no URL has been provided, 0L is
217      * returned. If the archive URL is not reachable or any other error occurs, Long.MAX_VALUE is returned.
218      *
219      * @return the timestamp of the last modification to the remote archive file.
220      */
221     public long getURLLastModificationDate()
222     {
223         if (getArchiveURL() == null) {
224             return 0;
225         }
226 
227         long lastmodified = Long.MAX_VALUE;
228         try {
229             URLConnection uc = getArchiveURL().openConnection();
230             lastmodified = uc.getLastModified();
231         } catch (IOException e) { // ignored
232         }
233         return lastmodified;
234     }
