16  package lu.softec.maven.mavenizer.mavenfile.internal;
17  
18  import java.io.File;
19  
20  import lu.softec.maven.mavenizer.mavenfile.MavenFile;
21  import lu.softec.maven.mavenizer.mavenfile.MavenFileSet;
22  
23  /**
24   * Internal mutable derived class used to construct a {@link MavenFile} and populates its dependencies during analysis
25   * or deserialisation.
26   */
27  class InternalMavenFile extends MavenFile
28  {
29      /**
30       * Constructs a new instance, populating all fields with provided values.
31       *
32       * @param file the file associated with the following coordinates
33       * @param groupId the groupId of the file
34       * @param artifactId the artifactId of the file
35       * @param version the version of the file
36       * @param classifier the classifier of the file
37       * @param deps the dependencies of the file
38       */
39      InternalMavenFile(File file, String groupId, String artifactId, String version, String classifier,
40          MavenFileSet deps)
41      {
42          this.file = file;
43          this.groupId = groupId;
44          this.artifactId = artifactId;
45          this.version = version;
46          this.classifier = classifier;
47          this.dependencies = (deps != null) ? deps : new InternalMavenFileSet();
48      }
49  
50      /**
51       * Add a new dependency
52       *
53       * @param file the dependency to add
54       * @return true if the dependency set has been changed
55       */
56      boolean addDependency(MavenFile file)
57      {
58          return ((InternalMavenFileSet) dependencies).add(file);
59      }
60  
61      /**
62       * Set the whole dependency set at once
63       *
64       * @param dependencies the new dependency set replacing any existing one
65       */
66      void setDependencies(MavenFileSet dependencies)
67      {
68          this.dependencies = dependencies;
69      }
70  }
