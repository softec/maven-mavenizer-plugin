16  package lu.softec.maven.mavenizer.analyzer.mavenasm;
17  
18  import java.io.IOException;
19  import java.io.InputStream;
20  
21  import org.objectweb.asm.ClassReader;
22  
23  import lu.softec.maven.mavenizer.analyzer.AbstractClassWalkListener;
24  import lu.softec.maven.mavenizer.analyzer.ClassWalkerRuntimeException;
25  
26  /**
27   * Abstract class extending {@link lu.softec.maven.mavenizer.analyzer.AbstractClassWalkListener} and providing basic
28   * functionalities to parse classes using a {@link org.objectweb.asm.ClassReader} and a {@link FileAwareClassVisitor}.
29   *
30   * Derived class should at least implements error of the ClassWalkListener interface for error reporting.
31   *
32   * WARNING! Derived class should ensure calling super for libraryWalkFileOpened and libraryWalkFileClosed overridden
33   * methods or provide accurate values by overriding getCurrentFile, getClassCount and incrementClassCount.
34   */
35  public abstract class AbstractClassWalkVisitorListener extends AbstractClassWalkListener
36  {
37      /**
38       * The class analyser visitor linked to this listener.
39       */
40      final FileAwareClassVisitor visitor;
41  
42      /**
43       * Construct an instance linked with the provided visitor.
44       *
45       * @param visitor the visitor that will be used to visit classes processed by this listener.
46       */
47      public AbstractClassWalkVisitorListener(FileAwareClassVisitor visitor)
48      {
49          this.visitor = visitor;
50      }
51  
52      /**
53       * Implement libraryWalkProcessClass of interface ClassWalkListener to visit class using a {@link
54       * org.objectweb.asm.ClassReader} and the provided {@link FileAwareClassVisitor}. Potential IOException are catched
55       * and passed to the listener error method.
56       *
57       * Before the visit, the visitor setFile and setClassPosition methods are called with values from getCurrentFile()
58       * and getClassCount(). After the visit, these methods are called again with values null and -1 respectively and
59       * than incrementClassCount is called.
60       *
61       * @param progress the approximative percentage of progress in the current walked file. For .class file, always
62       * 100%.
63       * @param in an input stream to read the class bytecode. This stream SHOULD NEVER be closed.
64       */
65      public void libraryWalkProcessClass(int progress, InputStream in)
66      {
67          try {
68              visitor.setFile(getCurrentFile());
69              visitor.setClassPosition(getClassCount());
70              new ClassReader(in).accept(visitor, ClassReader.SKIP_DEBUG ^ ClassReader.EXPAND_FRAMES);
71              visitor.setFile(null);
72              visitor.setClassPosition(-1);
73              incrementClassCount();
74          } catch (IOException e) {
75              throw new ClassWalkerRuntimeException(getCurrentFile(), e);
76          }
77      }
78  }
