200     public static class TimeRange
201     {
202         private long lower;

204         private long higher;

206         TimeRange(long lower, long higher)

208             this.lower = lower;
209             this.higher = higher;

223     /**
224      * Listener class to compute the modification time range of a list of file
225      */
226     private static class TimeRangeListener implements DirectoryWalkListener
227     {
228         private long lowerTime = Long.MAX_VALUE;
229 
230         private long higherTime = 0;
231 
232         public TimeRange getModificationTimeRange()
233         {
234             return new TimeRange(lowerTime, higherTime);
235         }
236 
237         public void directoryWalkStarting(File file)
238         {
239         }
240 
241         public void directoryWalkStep(int i, File file)
242         {
243             if (file.lastModified() < lowerTime) {
244                 lowerTime = file.lastModified();
245             }
246             if (file.lastModified() > higherTime) {
247                 higherTime = file.lastModified();
248             }
249         }
250 
251         public void directoryWalkFinished()
252         {
253         }
254 
255         public void debug(String s)
256         {
257         }
258     }

260     /**
261      * Returns the modifications time range of files under a given directory
262      *
263      * @param file the directory to be scanned
264      * @return the modifications time range of files under a given directory or null if file is not a directory
265      */
266     public static TimeRange getFilesModificationTimes(File file)
267     {
268         if (!file.isDirectory()) {
269             return new TimeRange(0, Long.MAX_VALUE);
270         }
271 
272         DirectoryWalker walker = new DirectoryWalker();
273         TimeRangeListener listener = new TimeRangeListener();
274 
275         walker.setBaseDir(file);
276         walker.addDirectoryWalkListener(listener);
277         walker.scan();
278         return listener.getModificationTimeRange();
279     }

303     /**
304      * Check that a file is in the build directory.
305      *
306      * @param file file to check
307      * @return true if the file is in the build directory.
308      */
309     public boolean isBuilt(File file)
310     {
311         if (file == null) {
312             return false;
313         }
314         try {
315             return file.getCanonicalPath().startsWith(getBuildDir().getCanonicalPath());
316         } catch (IOException e) {
317             return false;
318         }
319     }

321     /**
322      * Delete a directory from the build directory. If file is null, file does not exist, or the directory is not in the
323      * build directory, silently do nothing.
324      *
325      * @param file directory to be deleted
326      * @throws MojoExecutionException when an I/O error occurs during deletion
327      */
328     public void deleteTargetDirectory(File file) throws MojoExecutionException
329     {
330         if (!isBuilt(file)) {
331             return;
332         }
333         if (file.exists() && !file.isDirectory()) {
334             throw new IllegalArgumentException("File " + file.getAbsolutePath() + "is not a directory.");
335         }
336         try {
337             FileUtils.deleteDirectory(file);
338         } catch (IOException e) {
339             throw new MojoExecutionException("Unable to delete directory:" + file.getAbsolutePath(), e);
340         }
341     }
342
